# Alternate server configuration

server {
	listen 80;
	server_name test.puzzleduck.org;

	location /sub/ {
		root /var/www/puzzleduck.org/test-html;
		index index.html index.htm index.nginx-debian.html index.php;
	}

	location /php/ {
		root /var/www/puzzleduck.org/test-html;
		index index.html index.htm index.nginx-debian.html index.php;
	}

	location / {
		root /var/www/puzzleduck.org/test-html;
		index index.html index.htm index.nginx-debian.html index.php;
	}

}
